
CREATE TABLE tatuador 
( 
 id_tatuador INT AUTO_INCREMENT PRIMARY KEY,  
 idsenioridade INT NOT NULL,  
 email VARCHAR(150) NULL UNIQUE,  
 datacontratacao DATE NOT NULL,  
 cpf CHAR(11) NOT NULL UNIQUE,  
 nometatuador VARCHAR(100) NOT NULL,  
 telefone VARCHAR(20) NULL
); 

CREATE TABLE estilo 
( 
 id_estilo INT AUTO_INCREMENT PRIMARY KEY,  
 estilo VARCHAR(100) NOT NULL UNIQUE 
); 

CREATE TABLE senioridade 
( 
 codsenioridade INT AUTO_INCREMENT PRIMARY KEY,  
 nome VARCHAR(100)  NOT NULL
); 

CREATE TABLE agendamento 
( 
 id_agendamento INT AUTO_INCREMENT PRIMARY KEY,  
 idtatuador INT AUTO_INCREMENT PRIMARY KEY,  
 status VARCHAR(50) NOT NULL,  
 descricaotatuagem TEXT NULL,  
 valotatuagem DECIMAL(10,2) NOT NULL,  
 dataagendamento DATE NOT NULL,  
 horaagendamento TIME NOT NULL,  
 desenho_aprovado INT NOT NULL 
); 

CREATE TABLE cliente 
( 
 id_cliente INT AUTO_INCREMENT PRIMARY KEY,   
 telefone VARCHAR(20) NULL,  
 nome VARCHAR(100) NOT NULL,  
 cpf VARCHAR(11) NOT NULL UNIQUE,  
 datacadastro DATE NOT NULL,  
 cidade VARCHAR(100) NULL,  
 uf VARCHAR(2) NULL  
); 

CREATE TABLE possui 
( 
 id_estilo INT PRIMARY KEY,  
 id_tatuador INT PRIMARY KEY,  
 dataestilo DATE  
); 

CREATE TABLE Rel 
( 
 id_agendamento INT PRIMARY KEY,  
 id_cliente INT PRIMARY KEY  
); 

ALTER TABLE tatuador ADD FOREIGN KEY(idsenioridade) REFERENCES senioridade (idsenioridade)
ALTER TABLE agendamento ADD FOREIGN KEY(idtatuador) REFERENCES tatuador (idtatuador)
ALTER TABLE possui ADD FOREIGN KEY(id_estilo) REFERENCES estilo (id_estilo)
ALTER TABLE possui ADD FOREIGN KEY(id_tatuador) REFERENCES tatuador (id_tatuador)
ALTER TABLE Rel ADD FOREIGN KEY(id_agendamento) REFERENCES agendamento (id_agendamento)
ALTER TABLE Rel ADD FOREIGN KEY(id_cliente) REFERENCES cliente (id_cliente)
